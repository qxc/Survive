% Adds the name, image and description to the card.
% The input is name, image_name
\newcommand{\cardinfo}[2]{
		% Add the title with some shadow
		\node[text width=14.5cm, below right] at (\cardwidth*.2,\cardheight*.95) {
       \shadowtext{\parbox{8.5cm}{\begin{spacing}{3.6}\textbf{\hspace{-1.5cm}\uppercase{\Huge #1}}\end{spacing}}}
    };
		% Add the image to the card.
		\node at (\cardwidth*.5, \cardheight*.7) {\includegraphics[height = 6.75cm, width = 9cm, keepaspectratio]{#2}};
}

% Adds the description and boxes it.
\newcommand{\cardcontent}[1]{
	\node[draw, rectangle, ultra thick, text width=15.5cm, below] at (\cardwidth*.5, 0.5*\cardheight) {\textrm{\Huge#1}};
}
